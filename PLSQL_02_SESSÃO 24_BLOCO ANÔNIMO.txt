--DBMS_OUTPUT: PACKAGE QUE MOSTRA TESTO NA TELA, DEPURAÇÃO DE MENSAGENS AVISO E CONTAÚDO DE VARIÁVEIS
--HABILITA BUFFER DE SAÍDA DA PACKAGE DBMS_OUTPUT

SET SERVEROUTPUT ON 
DECLARE 
    VNumero1 NUMBER(11,2) := 500;
    VNumero2 NUMBER(11,2) := 400;
    vMedia   NUMBER(11,2);
BEGIN 
    vMedia := (VNumero1 + VNumero2) / 2;
    DBMS_OUTPUT.PUT_LINE('MEDIA = ' || vMedia);
END;
